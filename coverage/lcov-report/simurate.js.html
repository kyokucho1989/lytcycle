function getCountObject(countSize) {
  let count = {
    totalCount: countSize,
  };
  let JSobjectProp = anime({
    targets: count,
    totalCount: countSize,
    easing: "linear",
    round: 1,
    update: function () {
      var el = document.querySelector("#JSobjectProp pre");
      el.innerHTML = JSON.stringify(count);
    },
  });
 
  return JSobjectProp;
}
